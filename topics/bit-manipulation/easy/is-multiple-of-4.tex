\question Determine whether an integer is a multiple of 4 using only a mask and equality operators.

\ifprintanswers\else

\begin{solution}
\begin{lstlisting}
static boolean isMultipleOfFour(int x) {
    return 0 == (x & 3);
}
\end{lstlisting}
\end{solution}
