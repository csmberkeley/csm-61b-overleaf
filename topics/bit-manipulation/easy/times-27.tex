\question Write an expression equivalent to \texttt{x * 27} without using \texttt{*}, \texttt{/}, or \texttt{\%}.

\ifprintanswers\else
\begin{lstlisting}
static int times27(int x) {
    return ______________________________________________________
}
\end{lstlisting}
\fi

\begin{solution}
\begin{lstlisting}
static int times27(int x) {
    return (x << 4) + (x << 3) + (x << 1) + x;
}
\end{lstlisting}
\end{solution}
