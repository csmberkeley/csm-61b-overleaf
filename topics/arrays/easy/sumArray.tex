public int[] sums(int[] input) {
    int n = input.length;
    int output[] = new int[n];
    int total = 0;
    for (int i = 0; i < n; i++) {
        total += input[i]
    }
    for (int i = 0; i < n; i++){
        output[i] = total - input[i]
    }
    return output;
}

Solution 2 (without using subtraction):
public int[] sums(int[] input) {
    int n = input.length;
    int[] output = new int[n];
    
    output[0] = 0;
    for (int i = 1; i < n; i++) {
        output[i] = output[i - 1] + input[i - 1];
    }
    
    int right = 0;
    for (int i = n - 2; i >= 0; i--) {
        output[i] = output[i + 1] + right;
        right += input[i];
    }

    return output;
}
