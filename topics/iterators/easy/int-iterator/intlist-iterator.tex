\begin{lstlisting}
public class IntListIterator implements Iterator<Integer> {
    private IntNode node;
    public IntListIterator(IntNode list) {
        node = list;
    }
    public boolean hasNext() {
        return node != null;
    }
    public Integer next() {
        int value = node.first;
        node = node.rest;
        return value;
    }
}
\end{lstlisting}
